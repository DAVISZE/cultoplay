(Inicio.java =
public class Inicio {

    public static void main(String[] args) {

        System.out.println("=================================");
        System.out.println("   CULTOPLAY UPDATE SYSTEM");
        System.out.println("=================================");

        System.out.println("Versao carregada com sucesso!");
        System.out.println("Sistema iniciado.");

        Teste.mensagem();
        Utilidades.info();

    }

}
)

(Teste.java scripts/ =
public class Teste {

    public static void mensagem() {

        System.out.println("Executando modulo de teste...");
        System.out.println("Todos os arquivos foram atualizados.");

        for(int i = 1; i <= 5; i++) {
            System.out.println("Processo " + i + " OK");
        }

        System.out.println("Teste finalizado.");

    }

}
)

(Utilidades.java core/system/ =
public class Utilidades {

    public static void info() {

        System.out.println("=== SISTEMA ===");
        System.out.println("Updater conectado ao GitHub.");
        System.out.println("Arquivos sincronizados.");
        System.out.println("Sistema pronto.");

    }

}
)

scripts/ -> scripts do sistema
core/system/ -> núcleo do sistema
config/ -> arquivos de configuração
docs/ -> documentação
